Abstract Person {
        ClockIn()
        ClockOut()
        GetFired()
        Quit()
        GetPaycheck()
        CallIn()
}


Interface User{
LogIn()
        LogOut()
        UpdateDemographicsPersonal()
}


Class Manager extends Person implements User{
        Hire()
        Fire()
        GiveRaise()
        CreateEvaluation()
        ChangeEvaluation()
        UpdateDemographicsUniversal()
        ManagePayroll()
}


Class Employee extends Person implements User{
        DoWork()
        FillOutEvalutaion()
}
